def get_value(visible = true)
  obj, type = find_element(visible)
  object_not_found_exception(obj, type)
  result = obj.value
  unless result.blank?
    if result.is_int?
      result.to_i
    elsif result.is_float?
      result.to_f
    else
      result
    end
  end
end
